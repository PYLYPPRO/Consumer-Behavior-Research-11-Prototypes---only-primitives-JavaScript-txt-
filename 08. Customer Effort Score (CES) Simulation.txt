// --- Customer Effort Score (CES) Calculation in JavaScript ---

function calculateCES() {
    // 1. Collect Data
    const reg = parseInt(document.getElementById('registration_effort').value);
    const tech = parseInt(document.getElementById('tech_support_effort').value);
    const finance = parseInt(document.getElementById('financial_aid_effort').value);
    const advisor = parseInt(document.getElementById('advisor_meeting_effort').value);

    // 2. CES Calculation (Simple Average)
    const totalScore = reg + tech + finance + advisor;
    const combinedCES = (totalScore / 4).toFixed(2);
    
    // Low Effort (Good) is a score > 5.0
    const effortVerdict = (combinedCES >= 6) ? "Very Low Effort (Excellent)" : 
                          (combinedCES >= 4) ? "Medium Effort (Acceptable)" : 
                          "High Effort (Critical)";

    // Identify Pain Point
    const scores = { 'Registration': reg, 'Tech Support': tech, 'Financial Aid': finance, 'Advisor Meeting': advisor };
    let minScore = 7;
    let painPoint = "None identified";
    for (const key in scores) {
        if (scores[key] < minScore) {
            minScore = scores[key];
            painPoint = key;
        }
    }

    // 3. Output and Detailed Interpretation
    const resultDiv = document.getElementById('result_ces');

    let message = `<h2>CUSTOMER EFFORT SCORE (CES) REPORT</h2>`;
    message += `<p><strong>Combined CES (1-7 Scale): ${combinedCES}</strong></p>`;
    message += `<p><strong>Effort Verdict:</strong> ${effortVerdict}</p>`;
    
    message += `<h3>SERVICE OPTIMIZATION INSIGHT:</h3>`;
    message += `<p><strong>Lowest Rated Interaction (Pain Point):</strong> ${painPoint} (Score: ${minScore})</p>`;

    if (combinedCES < 4) {
        message += `<p><strong>CRITICAL FAILURE:</strong> The overall process is causing significant friction. **Action:** Launch a "Zero Effort" project focused on the ${painPoint} process to remove steps, streamline forms, or improve self-service options.</p>`;
    } else if (minScore < 4) {
        message += `<p><strong>TARGETED IMPROVEMENT:</strong> While overall effort is acceptable, the ${painPoint} process is a weak link. **Action:** Isolate the ${painPoint} department and conduct a deep dive (process mapping) to fix the bottleneck causing the high effort score.</p>`;
    } else {
        message += `<p><strong>MAINTAIN AND MONITOR:</strong> All scores are acceptable. **Action:** Continue monitoring, focusing the next survey wave on a different set of service interactions (e.g., library, dining, career placement).</p>`;
    }
    
    resultDiv.innerHTML = message;
}
</script>
