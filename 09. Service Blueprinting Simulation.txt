<script>
// --- Service Blueprinting Simulation in JavaScript ---

function analyzeServiceBlueprint() {
    // 1. Collect Data
    const fsAdmissions = parseInt(document.getElementById('front_stage_admissions').value);
    const bsDataEntry = parseInt(document.getElementById('back_stage_data_entry').value);
    const fsLMS = parseInt(document.getElementById('front_stage_lms_interface').value);
    const bsServer = parseInt(document.getElementById('back_stage_server_reliability').value);

    // 2. Calculate Gaps
    const gapAdmissions = fsAdmissions - bsDataEntry;
    const gapLMS = fsLMS - bsServer;

    // 3. Output and Detailed Interpretation
    const resultDiv = document.getElementById('result_blueprint');

    let message = `<h2>SERVICE BLUEPRINT ANALYSIS REPORT</h2>`;
    message += `<p><strong>Service Quality Scale:</strong> 1 (Poor) to 10 (Excellent)</p>`;
    
    message += `<h3>ADMISSIONS PROCESS ANALYSIS:</h3>`;
    message += `<ul>
        <li>Front-Stage Admissions: ${fsAdmissions}</li>
        <li>Back-Stage Data Entry: ${bsDataEntry}</li>
        <li><strong>Quality Gap: ${gapAdmissions}</strong></li>
    </ul>`;
    
    message += `<h3>LMS TECHNOLOGY ANALYSIS:</h3>`;
    message += `<ul>
        <li>Front-Stage LMS Usability: ${fsLMS}</li>
        <li>Back-Stage Server Reliability: ${bsServer}</li>
        <li><strong>Quality Gap: ${gapLMS}</strong></li>
    </ul>`;

    message += `<h3>OPERATIONAL CONCLUSION:</h3>`;
    
    if (Math.abs(gapAdmissions) > 4 && gapAdmissions > 0) {
        message += `<p><strong>CRITICAL GAP (Admissions):</strong> High Front-Stage rating (${fsAdmissions}) hides a Back-Stage failure (${bsDataEntry}). The customer-facing staff is excellent, but internal systems (Data Entry) are failing, creating errors downstream. **Action:** Invest in automation or mandatory staff training for backstage processes.</p>`;
    } else if (Math.abs(gapLMS) > 2 && gapLMS < 0) {
        message += `<p><strong>MISSED OPPORTUNITY (LMS):</strong> Server reliability is high (${bsServer}), but student usability is low (${fsLMS}). The technology is stable but poorly designed. **Action:** Dedicate resources to user experience (UX) design for the LMS interface. The reliable backend is not translating to student satisfaction.</p>`;
    } else {
        message += `<p><strong>GOOD ALIGNMENT:</strong> Front-Stage and Back-Stage processes are reasonably aligned. **Action:** Focus on incremental improvements (e.g., increasing both scores by 1 point) across the board for continuous quality assurance.</p>`;
    }
    
    resultDiv.innerHTML = message;
}
</script>
