        <strong>Method Explanation:</strong> Price Elasticity of Demand (E) measures the responsiveness of the quantity demanded (enrollment) to a change in price (tuition). E = (\Delta {Enrollment}) / (\Delta {Price}). If |E| > 1, demand is **Elastic** (price sensitive). If |E| < 1, demand is **Inelastic** (price insensitive).

<script>
// --- Price Elasticity of Demand Calculation in JavaScript ---

function calculateElasticity() {
    // 1. Collect Data
    const P1 = parseFloat(document.getElementById('initial_price').value);
    const P2 = parseFloat(document.getElementById('new_price').value);
    const Q1 = parseInt(document.getElementById('initial_enrollment').value);
    const Q2 = parseInt(document.getElementById('new_enrollment').value);

    // 2. Midpoint Formula for Elasticity (more accurate for large changes)
    // % Change in Quantity: (Q2 - Q1) / ((Q1 + Q2) / 2)
    // % Change in Price: (P2 - P1) / ((P1 + P2) / 2)
    
    const changeQ = (Q2 - Q1) / ((Q1 + Q2) / 2);
    const changeP = (P2 - P1) / ((P1 + P2) / 2);

    const elasticity = (changeP === 0) ? 0 : changeQ / changeP;
    const absoluteElasticity = Math.abs(elasticity).toFixed(2);
    
    // 3. Revenue Calculation
    const initialRevenue = P1 * Q1;
    const newRevenue = P2 * Q2;

    // 4. Output and Detailed Interpretation
    const resultDiv = document.getElementById('result_elasticity');

    let verdict = "";
    let action = "";
    if (absoluteElasticity > 1) {
        verdict = "ELASTIC (Price Sensitive)";
        action = "Price increases will significantly hurt enrollment volume and potentially reduce total revenue.";
    } else if (absoluteElasticity < 1) {
        verdict = "INELASTIC (Price Insensitive)";
        action = "Demand is relatively stable. Price increases will likely increase total revenue. However, monitor equity impact.";
    } else {
        verdict = "UNIT ELASTIC";
        action = "Total revenue remains unchanged. Consider non-price factors (quality, service) to gain advantage.";
    }

    let revenueChange = (newRevenue - initialRevenue) / initialRevenue * 100;

    let message = `<h2>PRICE ELASTICITY REPORT</h2>`;
    message += `<p><strong>Price Elasticity (E): ${absoluteElasticity}</strong> (|E|)</p>`;
    message += `<p><strong>Demand Classification:</strong> ${verdict}</p>`;
    message += `<p><strong>Revenue Change:</strong> ${revenueChange.toFixed(1)}% (From $${initialRevenue.toLocaleString()} to $${newRevenue.toLocaleString()})</p>`;
    
    message += `<h3>STRATEGIC FINANCIAL CONCLUSION:</h3>`;
    message += `<p><strong>Interpretation:</strong> For every 1% change in price, enrollment changes by ${absoluteElasticity}%.</p>`;
    message += `<p><strong>Actionable Insight:</strong> ${action}</p>`;

    if (revenueChange < 0) {
        message += `<p><strong>WARNING:</strong> The price change resulted in a ${Math.abs(revenueChange.toFixed(1))}% decrease in total revenue. **Recommendation:** Roll back the price increase immediately or drastically improve perceived quality/value to shift the demand curve.</p>`;
    } else if (revenueChange > 0) {
        message += `<p><strong>SUCCESS:</strong> The price change resulted in a ${revenueChange.toFixed(1)}% increase in total revenue. **Recommendation:** The market can bear the price; focus on optimizing the next price increment, carefully watching the equity implications.</p>`;
    }
    
    resultDiv.innerHTML = message;
}
</script>
