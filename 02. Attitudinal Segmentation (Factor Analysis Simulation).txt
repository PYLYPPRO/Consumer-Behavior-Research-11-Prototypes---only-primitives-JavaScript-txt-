// --- Attitudinal Segmentation Simulation in JavaScript ---

function segmentAttitude() {
    // 1. Collect Data
    const careerFocus = parseInt(document.getElementById('career_focus').value);
    const costSensitivity = parseInt(document.getElementById('cost_sensitivity').value);
    const academicCuriosity = parseInt(document.getElementById('academic_curiosity').value);
    const onlinePreference = parseInt(document.getElementById('online_preference').value);
    const socialLearning = parseInt(document.getElementById('social_learning').value);

    // 2. Segmentation Logic (Simulating Cluster Centers from Factor Analysis)
    let segment = { name: "Mid-Range / Undefined", description: "Default segment. Needs further behavioral data.", focus: "General institutional messaging." };

    if (careerFocus >= 8 && costSensitivity <= 4) {
        segment = {
            name: "Careerist/Pragmatist",
            description: "Driven by ROI and career outcomes. Less concerned with cost if value is proven.",
            focus: "Highlighting graduate employment rates, alumni salaries, and certification/licensure success stories."
        };
    } else if (academicCuriosity >= 7 && socialLearning >= 7 && careerFocus <= 5) {
        segment = {
            name: "Explorer/Community Builder",
            description: "Motivated by deep learning, intellectual challenge, and community interaction. Cost is secondary to experience.",
            focus: "Promoting research opportunities, faculty profiles, study abroad programs, and on-campus events."
        };
    } else if (costSensitivity >= 7 && onlinePreference >= 6) {
        segment = {
            name: "Cost-Sensitive/Flexibility Seeker",
            description: "Primary drivers are low cost, financial aid, and schedule flexibility. Often part-time or non-traditional students.",
            focus: "Information about scholarships, cost comparison with competitors, and promoting flexible/online course availability."
        };
    }

    // 3. Output and Detailed Interpretation
    const resultDiv = document.getElementById('result_attitude');

    let message = `<h2>ATTITUDINAL SEGMENTATION REPORT</h2>`;
    message += `<p><strong>Student's Scores:</strong> Career: ${careerFocus}, Cost Sensitivity: ${costSensitivity}, Curiosity: ${academicCuriosity}, Online: ${onlinePreference}, Social: ${socialLearning}</p>`;
    message += `<div class="segment-info">`;
    message += `<strong>ASSIGNED SEGMENT:</strong> ${segment.name}<br>`;
    message += `<strong>Description:</strong> ${segment.description}`;
    message += `</div>`;
    
    message += `<h3>STRATEGIC MARKETING CONCLUSION:</h3>`;
    message += `<p>To effectively acquire and retain this student, the marketing and advising communication must be tailored to their core psychological driver.</p>`;
    message += `<p><strong>Recommended Content Focus:</strong> ${segment.focus}</p>`;

    if (segment.name === "Careerist/Pragmatist") {
        message += `<p><strong>Retention Action:</strong> Link academic performance to career milestones. Ensure they meet with the Career Services team early in their program.</p>`;
    } else if (segment.name === "Explorer/Community Builder") {
        message += `<p><strong>Retention Action:</strong> Encourage enrollment in honors programs or interdisciplinary research clusters. Failure to provide intellectual challenge is a high churn risk.</p>`;
    } else if (segment.name === "Cost-Sensitive/Flexibility Seeker") {
        message += `<p><strong>Retention Action:</strong> Financial stability is key. Automate financial check-ins and highlight the value of their online degree compared to traditional formats.</p>`;
    }

    resultDiv.innerHTML = message;
}
